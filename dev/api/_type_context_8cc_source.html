80 const Type& TypeContext::find(const string& name, const SourceRange& src,
81  const PtrVec<Type>& params)
82 {
83  auto i = types.find(QualifiedName(name, params));
84  if (i != types.end())
85  return *i->second.get();
86 
87  if (not params.empty())
88  {
89  i = types.find(QualifiedName(name));
90  if (i != types.end())
91  {
92  const Type& unparam = *i->second;
93  Type *parameterised = unparam.Parameterise(params, src);
94 
95  if (parameterised and *parameterised)
96  Register(parameterised);
97 
98  return *parameterised;
99  }
100  }
101 
102  return nilType();
103 }
