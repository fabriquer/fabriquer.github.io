224 void EvalContext::DumpScope()
225 {
226  Bytestream& out = Bytestream::Debug("dag.scope");
227  size_t depth = 0;
228 
229  out
230  << Bytestream::Operator << "---------------------------\n"
231  << Bytestream::Definition << "Scopes (parent -> current):\n"
232  << Bytestream::Operator << "---------------------------\n"
233  ;
234 
235  for (auto scope = scopes_.begin(); scope != scopes_.end(); scope++)
236  {
237  const string indent(" ", depth);
238  for (auto i : *scope)
239  {
240  const string name(i.first);
241  const ValuePtr value(i.second);
242 
243  out << indent
244  << Bytestream::Operator << "- "
245  << Bytestream::Definition << name
246  << Bytestream::Operator << ": "
247  << *value
248  << Bytestream::Reset << "\n"
249  ;
250  }
251 
252  depth++;
253  }
254 
255  out
256  << Bytestream::Operator << "---------------------------\n"
257  << Bytestream::Reset
258  ;
259 }

261 ValueMap EvalContext::CopyCurrentScope()
262 {
263  ValueMap copy;
264 
265  for (auto i = scopes_.rbegin(); i != scopes_.rend(); i++)
266  {
267  // Unfortunately, std::copy() doesn't work here because
268  // it wants to copy *into* the string part of a const
269  // pair<string,SharedPtr<Value>>.
270  for (auto j : *i)
271  {
272  string name = j.first;
273  ValuePtr value = j.second;
274 
275  copy.emplace(name, value);
276  }
277  }
278 
279  return copy;
280 }

283 void EvalContext::Define(ScopedValueName& name, ValuePtr v)
284 {
285  assert(&name.stack_ == this);
286 
287  ValueMap& currentScope = CurrentScope();
288  if (currentScope.find(name.name_) != currentScope.end())

297 ValuePtr EvalContext::Lookup(const string& name)
298 {
299  Bytestream& dbg = Bytestream::Debug("dag.lookup");
300  dbg
301  << Bytestream::Action << "lookup "
302  << Bytestream::Literal << "'" << name << "'"
303  << Bytestream::Reset << "\n"
304  ;
305 
306  for (auto i = scopes_.rbegin(); i != scopes_.rend(); i++)
307  {
308  const ValueMap& scope = *i;
309 
310  auto value = scope.find(name);
311  if (value != scope.end())
312  {
313  assert(value->second);
314 
315  dbg
316  << Bytestream::Action << " found "
317  << Bytestream::Literal << "'" << name << "'"
318  << Bytestream::Operator << ": "
319  << *value->second
320  << Bytestream::Reset << "\n"
321  ;
322  return value->second;
323  }
324 
325  dbg
326  << " no "
327  << Bytestream::Literal << "'" << name << "'"
328  << Bytestream::Operator << ":"
329  ;
330 
331  for (auto& j : scope)
332  dbg << " " << Bytestream::Definition << j.first;
333 
334  dbg << Bytestream::Reset << "\n";
335  }
336 
337  // If we are looking for 'subdir' and haven't found it defined anywhere,
338  // provide the top-level source subdirectory ('').
339  if (name == ast::Subdirectory)
340  return builder_.File("", ValueMap(), ctx_.fileType());
341 
342  return nullptr;
343 }
