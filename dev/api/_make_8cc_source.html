115 void MakeBackend::Process(const dag::DAG& dag, Bytestream& out, ErrorReport::Report err)
116 {
117  MakeFormatter formatter;
118 
119  // Header comment:
120  out
121  << Bytestream::Comment
122  << "#\n"
123  << "# POSIX makefile generated by Fabrique\n"
124  << "#\n"
125  << Bytestream::Action << ".POSIX:\n"
126  << Bytestream::Action << ".SILENT:\n"
127  << "\n"
128  ;
129 
130 
131  // Variables:
132  out
133  << "#\n"
134  << "# Varables:\n"
135  << "#\n"
136  << Bytestream::Reset
137  ;
138 
139  for (auto& i : dag.variables())
140  {
141  if (dynamic_pointer_cast<Rule>(i.second))
142  continue;
143 
144  out
145  << Bytestream::Definition << i.first
146  << Bytestream::Operator << "=" << indent_
147  << Bytestream::Literal << formatter.Format(*i.second)
148  << Bytestream::Reset
149  << "\n"
150  ;
151  }
152  out << "\n";
153 
154 
155  //
156  // Explicitly-named pseudo-targets, in reverse declaration order.
157  //
158  // Make treats the first target as the automatic target, whereas fabfiles
159  // necessarily define subtargets first and top-level targets last.
160  //
161  const auto& topLevelTargets = dag.topLevelTargets();
162  for (auto i = topLevelTargets.rbegin(); i != topLevelTargets.rend(); i++)
163  {
164  const string& name = i->first;
165  const shared_ptr<Value>& target = i->second;
166 
167  if (dynamic_pointer_cast<Rule>(target))
168  continue;
169 
170  out
171  << Bytestream::Definition << name
172  << Bytestream::Operator << " : "
173  << Bytestream::Literal
174  << formatter.Format(*target)
175  << "\n"
176  ;
177  }
178 
179  out << "\n";
180 
181 
182  // 'clean' target:
183  out
184  << Bytestream::Definition << "clean"
185  << Bytestream::Operator << " :\n"
186  << Bytestream::Action << "\techo \"cleaning...\"\n"
187  << Bytestream::Action << "\trm -rf"
188  ;
189 
190  for (auto& f : dag.files())
191  if (f->generated())
192  out << " " << formatter.Format(*f);
193 
194  out << Bytestream::Reset << "\n\n";
195 
196 
197  // Output directories:
198  std::set<string> directories;
199  for (auto& f : dag.files())
200  {
201  assert(f);
202 
203  if (not f->generated())
204  continue;
205 
206  const string dir = f->directory();
207  if (dir.empty())
208  continue;
209 
210  directories.insert(dir);
211 
212  out
213  << Bytestream::Definition << formatter.Format(*f)
214  << Bytestream::Operator << " : | "
215  << Bytestream::Literal << dir
216  << Bytestream::Reset << "\n"
217  ;
218  }
219 
220 
221  // Build steps:
222  out
223  << Bytestream::Comment
224  << "#\n"
225  << "# Build steps:\n"
226  << "#\n"
227  << Bytestream::Reset
228  ;
229 
230  for (const string& dir : directories)
231  {
232  out
233  << Bytestream::Definition << dir
234  << Bytestream::Operator << ":\n"
235  << indent_ << Bytestream::Action << CreateDirCommand(dir)
236  << Bytestream::Reset << "\n"
237  ;
238  }
239 
240  StringMap<string> pseudoTargets;
241  size_t buildID = 0;
242  for (auto& i : dag.builds())
243  {
244  const dag::Build& build = *i;
245  const dag::Rule& rule = build.buildRule();
246 
247  //
248  // If the build generates multiple outputs, we need to define
249  // a pseudo-target that points to all outputs.
250  //
251 
252  Build::FileVec outputs = build.outputs();
253  if (outputs.size() > 1)
254  {
255  const string pseudoName =
256  rule.name() + "_" + std::to_string(++buildID);
257 
258  out
259  << Bytestream::Definition << pseudoName
260  << Bytestream::Operator << " :"
261  ;
262 
263  for (const shared_ptr<File>& f : outputs)
264  {
265  string name = formatter.Format(*f);
266  pseudoTargets[name] = pseudoName;
267  out << " " << name;
268  }
269 
270  out << "\n";
271  }
272 
273  //
274  // target1 target2 : input1 input2 | dependency1 dependency 2
275  //
276  out << Bytestream::Definition;
277 
278  for (const shared_ptr<File>& f : outputs)
279  out << formatter.Format(*f) << " ";
280 
281  out
282  << Bytestream::Operator << ":"
283  << Bytestream::Literal
284  ;
285 
286  for (const shared_ptr<File>& f : build.inputs())
287  out << " " << formatter.Format(*f);
288 
289 
290  //
291  // Build the command to be run (substitute $variables).
292  //
293  string command = formatter.Format(rule);
294 
295  string depfile;
296  auto dep = rule.arguments().find("depfile");
297  if (dep != rule.arguments().end())
298  depfile = formatter.Format(*dep->second);
299 
300  std::function<string (const shared_ptr<File>&)> shortName =
301  [](const shared_ptr<File>& f) { return f->relativeName(); };
302 
303  string description =
304  rule.hasDescription()
305  ? rule.description()
306  : rule.name()
307  + " [ "
308  + join(build.inputs(), shortName, " ")
309  + " ] => [ "
310  + join(build.outputs(), shortName, " ")
311  + " ]"
312  ;
313 
314  // TODO: once we start using C++14, clean up all of these duplicated
315  // replaceAll() calls with a generic lambda
316  for (auto& j : build.arguments())
317  {
318  const string name = j.first;
319  const string str = formatter.Format(*j.second);
320 
321  formatter.replaceAll(command, "${" + name + "}", str);
322  formatter.replaceAll(depfile, "${" + name + "}", str);
323  formatter.replaceAll(description, "${" + name + "}", str);
324  }
325 
326  formatter.replaceAll(command, "${in}", build.inputs());
327  formatter.replaceAll(depfile, "${in}", build.outputs());
328  formatter.replaceAll(description, "${in}", build.outputs());
329 
330  formatter.replaceAll(command, "${out}", build.inputs());
331  formatter.replaceAll(depfile, "${out}", build.outputs());
332  formatter.replaceAll(description, "${out}", build.outputs());
333 
334  out
335  << "\n"
336 
337  << indent_ << Bytestream::Action
338  << "echo \"" << description << "\"\n"
339 
340  << indent_ << Bytestream::Action
341  << "echo \"" << command << "\" >> build.log\n"
342 
343  << indent_ << Bytestream::Action << command << "\n"
344  ;
345 
346  // TODO: figure out a depfile mechanism that's compatible with
347  // BSD make
348  if (not depfile.empty())
349  {
350  switch (flavour_)
351  {
352  case Flavour::POSIX: // fall through
353  case Flavour::BSD:
354  err("dependency files currently only supported"
355  " for GNU makefiles (--format=gmake)",
356  build.source(),
357  ErrorReport::Severity::Warning);
358  break;
359 
360  case Flavour::GNU:
361  out
362  << Bytestream::Action << "-include "
363  << Bytestream::Literal << depfile
364  << Bytestream::Reset << "\n"
365  ;
366  break;
367  }
368  }
369 
370  out << "\n";
371  }
372 }

375 int MakeFormatter::replaceAll(string& haystack, const string& needle,
376  const string& replacement)
377 {
378  int replaced = 0;
379 
380  while (true)
381  {
382  size_t i = haystack.find(needle);
383  if (i == string::npos)
384  break;
385 
386  haystack.replace(i, needle.length(), replacement);
387  }
388 
389  return replaced;
390 }

392 int MakeFormatter::replaceAll(string& haystack, const string& pattern,
393  const Build::FileVec& files)
394 {
395  std::ostringstream oss;
396 
397  for (const shared_ptr<File>& f : files)
398  oss << Format(*f) << " ";
399 
400  string replacement = oss.str();
401  replacement = replacement.substr(0, replacement.length() - 1);
402 
403  return replaceAll(haystack, pattern, replacement);
404 }
