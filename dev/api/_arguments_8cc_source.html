155 Arguments* Arguments::Parse(int argc, char *argv[])
156 {
157  option::Stats stats(usage, argc - 1, argv + 1);
158  std::vector<option::Option> options(stats.options_max);
159  std::vector<option::Option> buffer(stats.buffer_max);
160  option::Parser opts(usage, argc - 1, argv + 1,
161  options.data(), buffer.data());
162 
163  if (opts.nonOptionsCount() > 1)
164  return NULL;
165 
166  const bool help = options[Help];
167 
168  const string input = opts.nonOptionsCount() == 1
169  ? opts.nonOption(0)
170  : "fabfile";
171 
172  const string output = options[OutputFile]
173  ? options[OutputFile].arg
174  : "-";
175 
176  const bool outputSpecified = options[OutputFile];
177 
178  const string format = options[Format]
179  ? options[Format].arg
180  : "ninja";
181 
182  const string debugPattern = options[DebugPattern]
183  ? options[DebugPattern].arg
184  : "none";
185 
186  return new Arguments {
187  help,
188  input,
189  output,
190  outputSpecified,
191  format,
192  options[ParseOnly],
193  options[PrettyPrintAST],
194  options[PrettyPrintDAG],
195  options[PrintOutput],
196  debugPattern
197  };
198 }

213 static ArgStatus NonEmpty(const option::Option& o, bool printErr)
214 {
215  ArgStatus req = Required(o, printErr);

229 static ArgStatus IsOutputFormat(const Option& opt, bool printErr)
230 {
231  ArgStatus basic = NonEmpty(opt, printErr);
232  if (basic != ARG_OK)
233  return basic;
234 
235  const string arg(opt.arg);
236 
237  for (const char** const& format : formatStrings)
238  if (arg == format[0])
239  return ARG_OK;
240 
241  return ARG_ILLEGAL;
242 }
