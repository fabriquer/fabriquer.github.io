231 bool DAGBuilder::Enter(const ast::Action& a)
232 {
233  string command;
234  ValueMap arguments;
235 
236  if (a.arguments().size() < 1)
237  throw SemanticException("Missing action arguments", a.source());
238 
239  for (ConstPtr<ast::Argument>& arg : a.arguments())
240  {
241  // Flatten the argument and convert it to a string.
242  shared_ptr<Value> value = flatten(arg->getValue());
243 
244  // The only keyword-less argument to action() is its command.
245  if (not arg->hasName() or arg->getName().name() == "command")
246  {
247  if (not command.empty())
248  throw SemanticException(
249  "Duplicate command", arg->source());
250 
251  command = value->str();
252  continue;
253  }
254 
255  shared_ptr<Value> v(new String(value->str(), stringTy,
256  arg->source()));
257  arguments.emplace(arg->getName().name(), v);
258  }
259 
260  currentValue.emplace(Rule::Create(currentValueName.top(),
261  command, arguments, a.type()));
262 
263  return false;
264 }

278 bool DAGBuilder::Enter(const ast::BinaryOperation& o)
279 {
280  shared_ptr<Value> lhs = flatten(o.getLHS());
281  shared_ptr<Value> rhs = flatten(o.getRHS());
282 
283  assert(lhs and rhs);
284 
285  shared_ptr<Value> result;
286  switch (o.getOp())
287  {
288  case ast::BinaryOperation::Add:
289  result = lhs->Add(rhs);
290  break;
291 
292  case ast::BinaryOperation::Prefix:
293  result = rhs->PrefixWith(lhs);
294  break;
295 
296  case ast::BinaryOperation::ScalarAdd:
297  if (lhs->canScalarAdd(*rhs))
298  result = lhs->ScalarAdd(rhs);
299 
300  else if (rhs->canScalarAdd(*lhs))
301  result = rhs->ScalarAdd(lhs);
302 
303  else
304  throw SemanticException(
305  "invalid types for addition",
306  SourceRange::Over(lhs.get(), rhs.get())
307  );
308  break;
309 
310  case ast::BinaryOperation::And:
311  result = lhs->And(rhs);
312  break;
313 
314  case ast::BinaryOperation::Or:
315  result = lhs->Or(rhs);
316  break;
317 
318  case ast::BinaryOperation::Xor:
319  result = lhs->Xor(rhs);
320  break;
321 
322  case ast::BinaryOperation::Invalid:
323  break;
324  }
325 
326  assert(result);
327  currentValue.emplace(result);
328 
329  return false;
330 }

346 bool DAGBuilder::Enter(const ast::Call& call) { return false; }
347 void DAGBuilder::Leave(const ast::Call& call)
348 {
349  const ast::SymbolReference& ref = call.target();
350  const string name = ref.getName().name();
351  auto& target = dynamic_cast<const ast::Callable&>(ref.definition());
352 
353  ValueMap args;
354  for (auto& i : target.NameArguments(call.arguments()))
355  args[i.first] = std::move(flatten(*i.second));
356 
357 
358  //
359  // The target must be an action or a function.
360  //
361  if (auto rule = dynamic_pointer_cast<Rule>(getNamedValue(name)))
362  {
363  // Builds need the parameter types, not just the argument types.
364  ConstPtrMap<Type> paramTypes;
365  auto& params = target.parameters();
366  for (auto& a : args)
367  {
368  const string& name = a.first;
369 
370  auto i = params.begin();
371  for ( ; i != params.end(); i++)
372  if ((*i)->getName().name() == name)
373  break;
374 
375  assert(i != params.end());
376  const Type& t = (*i)->type();
377  paramTypes[name] = &t;
378  }
379 
380  shared_ptr<Build> build(
381  Build::Create(rule, args, paramTypes, call.source()));
382 
383  builds_.push_back(build);
384  currentValue.push(build);
385  }
386  else
387  {
388  auto& fn = dynamic_cast<const ast::Function&>(ref.definition());
389 
390  //
391  // We evaluate the function with the given arguments by
392  // putting these names into the local scope and then
393  // entering the function's CompoundExpr.
394  //
395  ValueMap& scope = EnterScope("fn eval");
396 
397  for (auto& i : args)
398  scope[i.first] = i.second;
399 
400  currentValue.emplace(flatten(fn.body()));
401  ExitScope();
402  }
403 }

580 void DAGBuilder::Leave(const ast::Scope&)
581 {
582  ValueMap scopedSymbols = ExitScope();
583 
584  // We only save top-level values if we are, in fact, at the top level.
585  if (not scopes.empty())
586  return;
587 
588  const string scopeName = join(this->scopeName, ".");
589 
590  for (auto symbol : scopedSymbols)
591  {
592  const string name = join(scopeName, symbol.first, ".");
593  shared_ptr<Value>& v = symbol.second;
594 
595  if (dynamic_pointer_cast<File>(v)
596  or dynamic_pointer_cast<Build>(v))
597  {
598  // Do nothing: files and builds have already been
599  // added to files_ or builds_, respectively.
600  }
601  else if (auto rule = dynamic_pointer_cast<Rule>(v))
602  rules_[name] = rule;
603 
604  else if (auto target = dynamic_pointer_cast<Target>(v))
605  targets_[name] = target;
606 
607  else
608  variables_[name] = v;
609  }
610 }

669 void DAGBuilder::Leave(const ast::Value& v)
670 {
671  // Things like function definitions will never be flattened.
672  if (currentValue.empty())
673  return;
674 
675  ValueMap& currentScope = CurrentScope();
676 
677  Bytestream& dbg = Bytestream::Debug("dag.scope");
678  dbg
679  << Bytestream::Action << "defining "
680  << Bytestream::Literal << "'" << currentValueName.top() << "'"
681  << Bytestream::Operator << ":"
682  ;
683 
684  shared_ptr<Value> val = std::move(currentValue.top());
685  currentValue.pop();
686 
687  const string name = currentValueName.top();
688  currentValueName.pop();
689 
690 
691  //
692  // If the right-hand side is a build, file or list of files,
693  // convert to a named target (files and builds are already in the DAG).
694  //
695  if (auto build = dynamic_pointer_cast<Build>(val))
696  val.reset(Target::Create(name, build));
697 
698  else if (auto file = dynamic_pointer_cast<File>(val))
699  val.reset(Target::Create(name, file));
700 
701  else if (auto list = dynamic_pointer_cast<List>(val))
702  {
703  if (list->type().isListOf(fileType))
704  val.reset(Target::Create(name, list));
705  }
706 
707 
708  currentScope.emplace(name, std::move(val));
709 
710  for (auto& i : currentScope)
711  dbg << Bytestream::Definition << " " << i.first;
712 
713  dbg << Bytestream::Reset << "\n";
714 }
