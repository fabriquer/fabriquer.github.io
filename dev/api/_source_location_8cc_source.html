181 void SourceRange::PrettyPrint(Bytestream& out, size_t /*indent*/) const
182 {
183  out
184  << Bytestream::Filename << begin.filename
185  << Bytestream::Operator << ":"
186  ;
187 
188  // The end column is the first character in the next token; don't
189  // report this when printing out the current location.
190  size_t endcol = end.column ? end.column - 1 : 0;
191 
192  if (begin.line == end.line)
193  {
194  out
195  << Bytestream::Line << begin.line
196  << Bytestream::Operator << ":"
197  << Bytestream::Column << begin.column
198  ;
199 
200  if (endcol != begin.column)
201  out
202  << Bytestream::Operator << "-"
203  << Bytestream::Column << endcol
204  ;
205  }
206  else
207  out
208  << Bytestream::Line << begin.line
209  << Bytestream::Operator << ":"
210  << Bytestream::Column << begin.column
211  << Bytestream::Operator << "-"
212  << Bytestream::Line << end.line
213  << Bytestream::Operator << ":"
214  << Bytestream::Column << endcol
215  ;
216 
217  out << Bytestream::Reset;
