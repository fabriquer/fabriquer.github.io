67 const Type& Scope::Lookup(const Identifier& id) const
68 {
69  const std::string& name = id.name();
70 
71  // Special case: 'args' is a reserved name derived from external arguments
72  // (command-line or via an import() expression).
73  if ((name == ast::Arguments) and arguments_.valid())
74  return arguments_;
75 
76  // Another special case: 'subdir' is a file.
77  if (name == ast::Subdirectory)
78  return arguments_.context().fileType();
79 
80  auto i = symbols_.find(name);
81  if (i != symbols_.end())
82  return i->second;
83 
84  if (parent_)
85  return parent_->Lookup(id);
86 
87  return nil_;
88 }

124 void Scope::Register(const Identifier& id, const Type& t)
125 {
126  Bytestream::Debug("ast.scope")
127  << Bytestream::Action << "scope"
128  << Bytestream::Operator << " <- "
129  << Bytestream::Type << "symbol"
130  << Bytestream::Operator << " = " << id
131  << Bytestream::Operator << ": " << t
132  << "\n"
133  ;
134 
135  if (symbols_.find(id.name()) != symbols_.end())
136  throw SyntaxError("name '" + id.name() + "' already defined",
137  id.source());
138 
139  symbols_.emplace(id.name(), t);
140 }
