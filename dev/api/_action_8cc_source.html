55 size_t Count(const UniqPtrVec<T>& values, TypePredicate predicate)
56 {
57  size_t count = 0;
58 
59  for (auto& v : values)
60  {
61  const Type& t = v->type();
62 
63  // If we have a list of input files, treat it as "more than one".
64  if (t.isOrdered())
65  {
66  auto& seqType = dynamic_cast<const SequenceType&>(t);
67  if (predicate(seqType.elementType()))
68  count += 2;
69  }
70  else if (predicate(v->type()))
71  {
72  count += 1;
73  }
74  }
75 
76  return count;
77 }

80 Action* Action::Create(UniqPtrVec<Argument>& args,
81  UniqPtr<UniqPtrVec<Parameter>>& params,
82  const SourceRange& src, TypeContext& ctx)
83 {
84  UniqPtrVec<Parameter> parameters;
85  if (params)
86  {
87  // Verify that all file parameters are either inputs or outputs.
88  for (auto& p : *params)
89  {
90  const Type& t = p->type();
91  if (t.isFile())
92  {
93  auto& file = dynamic_cast<const FileType&>(t);
94  if (not file.isInputFile()
95  and not file.isOutputFile())
96  throw WrongTypeException(
97  "file[in|out]", t, p->source());
98  }
99  }
100 
101  parameters = std::move(*params);
102  }
103 
104  const Type& file = ctx.fileType();
105  const Type& fileList = ctx.fileListType();
106 
107  const FunctionType& type = ctx.functionType(
108  Count(parameters, FileType::isInput) == 1 ? file : fileList,
109  Count(parameters, FileType::isOutput) == 1 ? file : fileList);
110 
111  return new Action(args, parameters, type, src);
112 }

173 dag::ValuePtr Action::evaluate(EvalContext& ctx) const
174 {
175  string command;
176  dag::ValueMap arguments;
177 
178  if (args_.size() < 1)
179  throw SemanticException("Missing action arguments", source());
180 
181  for (const UniqPtr<Argument>& arg : args_)
182  {
183  // Evaluate the argument as a string.
184  dag::ValuePtr value = arg->getValue().evaluate(ctx);
185 
186  // The only keyword-less argument to action() is its command.
187  if (not arg->hasName() or arg->getName().name() == "command")
188  {
189  if (not command.empty())
190  throw SemanticException(
191  "Duplicate command", arg->source());
192 
193  command = value->str();
194  continue;
195  }
196 
197  dag::ValuePtr v(
198  new dag::String(value->str(),
199  type().context().stringType(),
200  arg->source())
201  );
202  arguments.emplace(arg->getName().name(), v);
203  }
204 
205  SharedPtrVec<dag::Parameter> parameters;
206  for (const UniqPtr<Parameter>& p : this->parameters())
207  {
208  // Ensure that files are properly tagged as input or output.
209  FileType::CheckFileTags(p->type(), p->source());
210 
211  std::shared_ptr<dag::Parameter> param =
212  std::dynamic_pointer_cast<dag::Parameter>(
213  p->evaluate(ctx));
214 
215  assert(param);
216  parameters.emplace_back(param);
217  }
218 
219  return ctx.builder().Rule(command, arguments, parameters, type(), source());
220 }
