135 dag::ValuePtr BinaryOperation::evaluate(EvalContext& ctx) const
136 {
137  dag::ValuePtr lhs = lhs_->evaluate(ctx);
138  dag::ValuePtr rhs = rhs_->evaluate(ctx);
139  assert(lhs and rhs);
140 
141  switch (op_)
142  {
143  case Add: return lhs->Add(rhs);
144  case Prefix: return rhs->PrefixWith(lhs);
145  case And: return lhs->And(rhs);
146  case Or: return lhs->Or(rhs);
147  case Xor: return lhs->Xor(rhs);
148  case Equal: return lhs->Equals(rhs);
149  case NotEqual: return lhs->Equals(rhs)->Negate(source());
150 
151  case ScalarAdd:
152  if (lhs->canScalarAdd(*rhs))
153  return lhs->ScalarAdd(rhs);
154 
155  else if (rhs->canScalarAdd(*lhs))
156  return rhs->ScalarAdd(lhs);
157 
158  else
159  throw SemanticException(
160  "invalid types for addition", source());
161 
162  case Invalid:
163  throw SemanticException("invalid operation", source());
164  }
165 
166  assert(false && "unreachable");
167 }

179 const fabrique::Type& BinaryOperation::ResultType(const Type& lhs, const Type& rhs,
180  Operator op, SourceRange& loc)
181 {
182  switch (op)
183  {
184  case Add:
185  if (auto& t = lhs.onAddTo(rhs))
186  return t;
187 
188  if (auto& t = rhs.onAddTo(lhs))
189  return t;
190 
191  break;
192 
193  case Prefix:
194  if (auto& t = rhs.onPrefixWith(lhs))
195  return t;
196 
197  break;
198 
199  case ScalarAdd:
200  {
201  if (lhs.isOrdered() and lhs.typeParamCount() == 1
202  and lhs[0].onAddTo(rhs))
203  {
204  return lhs.Map(AddElementTypeTo(rhs), loc);
205  }
206 
207  if (rhs.isOrdered() and rhs.typeParamCount() == 1
208  and rhs[0].onAddTo(lhs))
209  {
210  return rhs.Map(AddElementTypeTo(lhs), loc);
211  }
212 
213  break;
214  }
215 
216  case And:
217  case Or:
218  case Xor:
219  case Equal:
220  case NotEqual:
221  if (lhs == rhs)
222  return lhs.context().booleanType();
223  break;
224 
225  case Invalid:
226  break;
227  }
228 
229  throw SemanticException("incompatible types: "
230  + lhs.str() + " vs " + rhs.str(), loc);
231 }
