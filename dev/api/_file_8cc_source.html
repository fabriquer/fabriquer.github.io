146 ValuePtr File::field(const string& name) const
147 {
148  TypeContext& ctx = type().context();
149  ValuePtr val;
150 
151  if (name == ast::Basename)
152  val.reset(new String(BaseName(filename_), ctx.stringType(), source()));
153 
154  else if (name == ast::Extension)
155  val.reset(new String(FileExtension(filename_), ctx.stringType(),
156  source()));
157 
158  else if (name == ast::FileName)
159  val.reset(new String(FilenameComponent(filename_), ctx.stringType(),
160  source()));
161 
162  else if (name == ast::Generated)
163  val.reset(new Boolean(generated_, ctx.booleanType(), source()));
164 
165  else if (name == ast::Name)
166  val.reset(new String(fullName(), ctx.stringType(), source()));
167 
168  else if (name == ast::Subdirectory)
169  val.reset(File::Create(subdirectory(), ValueMap(), ctx.fileType(),
170  source()));
171 
172  else
173  {
174  auto i = attributes_.find(name);
175  if (i != attributes_.end())
176  val = i->second;
177  }
178 
179  return val;
180 }
