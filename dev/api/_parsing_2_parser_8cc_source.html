75 const Type& Parser::ParseDefinitions(const std::vector<string>& definitions)
76 {
77  if (definitions_)
78  throw UserError("arguments already defined");
79 
80  Type::NamedTypeVec args;
81  const Type& nil = ctx_.nilType();
82 
83  UniqPtr<Scope> scope { new Scope(nullptr, "definitions", nil, ctx_) };
84 
85  for (const string& d : definitions)
86  {
87  std::istringstream input(d + ";");
88  UniqPtr<ast::Scope> definitionTree { ParseFile(input, nil) };
89  if (not definitionTree)
90  throw UserError("invalid definition '" + d + "'");
91 
92  for (UniqPtr<ast::Value>& value : definitionTree->TakeValues())
93  {
94  const string name = value->name().name();
95 
96  if (name == ast::Arguments)
97  continue;
98 
99  Bytestream::Debug("parser.cli.defines")
100  << Bytestream::Action << "Parsed definition"
101  << Bytestream::Operator << ": "
102  << *value
103  << Bytestream::Reset << "\n"
104  ;
105 
106  args.emplace_back(name, value->type());
107  scope->Take(value);
108  }
109 
110  definitionTree.reset();
111  }
112 
113  definitions_.swap(scope);
114 
115  return ctx_.structureType(args);
116 }

119 UniqPtr<Scope> Parser::ParseFile(std::istream& input, const Type& args,
120  string name, StringMap<string> builtins,
121  SourceRange openedFrom)
122 {
123  Bytestream& dbg = Bytestream::Debug("parser.file");
124  dbg
125  << Bytestream::Action << "Parsing"
126  << Bytestream::Type << " file"
127  << Bytestream::Operator << " '"
128  << Bytestream::Literal << name
129  << Bytestream::Operator << "'"
130  ;
131 
132  if (args.valid())
133  {
134  dbg
135  << Bytestream::Reset << " with "
136  << Bytestream::Definition << "args"
137  << Bytestream::Operator << ": "
138  ;
139 
140  for (auto& i : args.fields())
141  dbg
142  << Bytestream::Definition << i.first
143  << Bytestream::Operator << ":"
144  << i.second
145  ;
146  }
147 
148  dbg << Bytestream::Reset << "\n";
149 
150  if (not name.empty())
151  {
152  const string relativeName =
153  (name.find(srcroot_) == 0)
154  ? name.substr(srcroot_.length() + 1)
155  : name;
156 
157  files_.push_back(relativeName);
158  }
159 
160  lexer_.PushFile(input, name);
161  EnterScope(name, args);
162 
163  // Define builtin strings like srcroot and builtroot.
164  for (std::pair<string,string> i : builtins)
165  if (not Builtin(i.first, i.second, openedFrom))
166  throw AssertionFailure("Builtin('" + i.first + "')",
167  "failed to create builtin");
168 
169  int result = yyparse(this);
170 
171  lexer_.PopFile();
172 
173  if (result != 0)
174  return nullptr;
175 
176  return ExitScope();
177 }

396  UniqPtr<Identifier>& field)
397 {
398  if (not structure or not field)
399  return nullptr;
400 
401  const string& name = field->name();
402 
403  const Type& structType = structure->type();
404  if (not structType.hasFields())
405  throw SemanticException(
406  "value of type '" + structType.str() + "' does not have fields",
407  structure->source());
408 
409  Type::TypeMap fieldTypes { structType.fields() };
410  auto i = fieldTypes.find(name);
411  if (i == fieldTypes.end())
412  // TODO: more details (e.g. legal fields)
413  throw SemanticException("no such field", field->source());
414 
415  const Type& fieldType = i->second;
416 
417  if (field->isTyped() and field->type() != fieldType)

419  else

422  return new class FieldAccess(structure, field);

426 FieldQuery* Parser::FieldQuery(UniqPtr<Expression>& structure,
427  UniqPtr<Identifier>& field, UniqPtr<Expression>& def,
428  SourceRange src)
429 {
430  if (not structure or not field)
431  return nullptr;
432 
433  const Type& structType = structure->type();
434  if (not structType.hasFields())
435  throw SemanticException(
436  "value of type '" + structType.str() + "' does not have fields",
437  structure->source());
438 
439  Type::TypeMap structFields = structType.fields();
440  auto i = structFields.find(field->name());
441  if (i != structFields.end())
442  {
443  const Type& fieldType = i->second;
444  const Type& initializerType = def->type();
445 
446  if (not fieldType.isSubtype(initializerType)
447  and not initializerType.isSubtype(fieldType))
448  throw WrongTypeException(initializerType, fieldType, src);
449  }
450 
451  const Type& t =
452  i == structFields.end()
453  ? def->type()
454  : Type::GetSupertype(def->type(), i->second)
455  ;
456 
457  return new class FieldQuery(structure, field, def, t, src);
458 }

883 {
884  if (not id or not e)
885  return false;
886

890  return false;

898  return false;

902 
903  if (id->isTyped() and !e->type().isSubtype(id->type()))
904  {
905  ReportError("assigning " + e->type().str()
906  + " to identifier of type " + id->type().str(),
907  range);

915  + t.str() + "')", range);
916  return false;
