136 UniqPtr<DAG> DAGBuilder::dag(const vector<string>& topLevelTargets) const
137 {
138  //
139  // Ensure all files are unique.
140  //
141  SharedPtrVec<class File> f = files_;
142  std::sort(f.begin(), f.end(), File::LessThan);
143  f.erase(std::unique(f.begin(), f.end(), File::Equals), f.end());
144 
145 
146  //
147  // Find top-level targets.
148  //
149  vector<DAG::BuildTarget> top;
150  for (const string& name : topLevelTargets)
151  {
152  auto t = targets_.find(name);
153  if (t != targets_.end())
154  top.emplace_back(name, t->second);
155  }
156 
157 
158  return UniqPtr<DAG>(new ImmutableDAG(ctx_.buildroot(), ctx_.srcroot(),
159  f, builds_, rules_, variables_, targets_, top));
160 }

163 ValuePtr DAGBuilder::AddRegeneration(const Arguments& commandLineArgs,
164  const vector<string>& inputFiles,
165  const vector<string>& outputFiles)
166 {
167  shared_ptr<Value> Nothing;
168  const SourceRange& Nowhere = SourceRange::None();
169 
170  TypeContext& t = ctx_.types();
171  const FileType& inputFileType = t.inputFileType();
172  const Type& inputType = t.listOf(inputFileType, Nowhere);
173  const FileType& outputType = t.outputFileType();
174  const Type& buildType = t.functionType(inputType, outputType);
175 
176  //
177  // First, construct the rule that regenerates output:file[out]
178  // given input:list[file[in]].
179  //
180  ValueMap ruleArgs;
181  ruleArgs["description"] = String("Regenerating ${output}");
182 
183  SharedPtrVec<Parameter> params;
184  params.emplace_back(new Parameter("rootInput", inputFileType, Nothing));
185  params.emplace_back(new Parameter("otherInputs", inputType, Nothing));
186  params.emplace_back(
187  new Parameter("output", t.listOf(outputType, Nowhere), Nothing));
188 
189  shared_ptr<dag::Rule> rule;
190  {
191  const string Name = Rule::RegenerationRuleName();
192  const string Command =
193  commandLineArgs.executable
194  + Arguments::str(commandLineArgs)
195  + " ${rootInput}"
196  ;
197 
198  ValuePtr r = Rule(Name, Command, ruleArgs, params, buildType);
199  rule = dynamic_pointer_cast<class Rule>(r);
200  }
201  assert(rule);
202 
203 
204  //
205  // Now, construct the build step that drives the rule above in order
206  // to actually generate the build file.
207  //
208  shared_ptr<dag::File> rootInput;
209  SharedPtrVec<dag::File> otherInputs;
210  for (const string& name : inputFiles)
211  {
212  shared_ptr<dag::File> file = dynamic_pointer_cast<class File>(
213  File(name, ValueMap(), inputFileType, Nowhere));
214 
215  assert(file);
216 
217  if (rootInput)
218  otherInputs.push_back(file);
219  else
220  rootInput = file;
221  }
222 
223  SharedPtrVec<Value> outputs;
224  for (const string& output : outputFiles)
225  outputs.push_back(File(output, ValueMap(), outputType));
226 
227  ValueMap args;
228  args["rootInput"] = rootInput;
229  args["otherInputs"].reset(List::of(otherInputs, Nowhere, t));
230 
231  args["output"].reset(List::of(outputs, Nowhere, t));
232 
233  return Build(rule, args, Nowhere);
234 }
