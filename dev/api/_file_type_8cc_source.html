79 void FileType::CheckFileTags(const Type& t, SourceRange src)
80 {
81  if (t.isFile())
82  {
83  auto& file = dynamic_cast<const FileType&>(t);
84  if (not file.isInputFile() and not file.isOutputFile())
85  throw WrongTypeException("file[in|out]", file, src);

87  else if (t.isOrdered())

142 bool FileType::isSubtype(const Type& candidateSupertype) const
143 {
144  if (not candidateSupertype.isFile())
145  return false;
146 
147  auto& t = dynamic_cast<const FileType&>(candidateSupertype);
148  if (not t)
149  return false;

153 
154  // Can always pass a file to a file[in] or file[out] parameter.

195 Type* FileType::Parameterise(const PtrVec<Type>& params, const SourceRange& src) const
196 {
197  assert(tag_ == Tag::None);
198  assert(params.size() == 1);
199 
200  const std::string name = params[0]->name();
201 
202  if (name == InTagName)
203  return new FileType(Tag::Input, params, arguments_, context());
204 
205  else if (name == OutTagName)
206  return new FileType(Tag::Output, params, arguments_, context());
207 
208  throw SemanticException("invalid file tag '" + name + "'", src);
209 }
