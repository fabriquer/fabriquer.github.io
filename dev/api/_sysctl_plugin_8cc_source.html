146 static string SysctlName(const ValueMap& args)
147 {
148  // If the user didn't pass a 'name' argument of type string,
149  // dag::Callable ought to have caught it.
150  assert(args.size() == 1);
151  assert(args.find("name") != args.end());
152 
153  ValuePtr name = args.find("name")->second;
154  assert(name);
155 
156  return name->str();
157 }

160 static ValuePtr StringSysctl(const ValueMap& /*scope*/, const ValueMap& args,
161  DAGBuilder& builder, SourceRange src)
162 {
163  const string name = SysctlName(args);
164  const char *rawName = name.c_str();
165 
166  size_t len;
167  if (sysctlbyname(rawName, NULL, &len, NULL, 0))
168  throw PosixError(
169  "error querying size of '" + name + "' sysctl");
170 
171  UniqPtr<char> buffer(new char[len]);
172  if (sysctlbyname(rawName, buffer.get(), &len, NULL, 0))
173  throw PosixError(
174  "error retrieving '" + name + "' via sysctlbyname()");
175 
176  return builder.String(buffer.get(), src);
177 }
