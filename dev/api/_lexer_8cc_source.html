137 void Lexer::SetComment(YYSTYPE *yyunion, bool includesNewline)
138 {
139  string s = currentToken_.str();
140  SourceRange src = currentToken_.source();
141 
142  if (includesNewline)
143  {
144  s = s.substr(0, s.length() - 1);
145 
146  src.begin.line--;
147  src = range(s.data(), s.length(), src.begin);
148  yycolumn = 1;
149  }
150 
151  yyunion->token = new Token(s, src);
152  Token &t = *yyunion->token;
153 
154  Bytestream::Debug("lex.comment")
155  << Bytestream::Action << "lexed "
156  << Bytestream::Type << "comment"
157  << Bytestream::Operator << ": '"
158  << Bytestream::Comment << t.str()
159  << Bytestream::Operator << "' @ " << t.source()
160  << Bytestream::Reset << "\n"
161  ;
162 }

218 static SourceRange range(const char *text, size_t len, SourceLocation begin)
219 {
220  size_t line = begin.line;
221  size_t column = begin.column;
222 
223  for (size_t i = 0; i < len; i++)
224  {
225  if (text[i] == '\n')
226  {
227  line++;
228  column = 1;
229  }
230  else
231  column++;
232  }
233 
234  SourceLocation end(begin.filename, line, column);
235 
236  return SourceRange(begin, end);
237 }
