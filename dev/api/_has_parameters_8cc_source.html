65 void HasParameters::CheckArguments(const UniqPtrVec<Argument>& args,
66  const SourceRange& src) const
67 {
68  const auto& names = parameterNames();
69 
70  for (auto& a : args)
71  {
72  if (a->hasName())
73  {
74  const string name(a->getName().name());
75  if (find(names.begin(), names.end(), name) == names.end())
76  throw SemanticException(
77  "invalid parameter", a->source());
78  }
79  }
80 
81  StringMap<const Argument*> namedArguments = NameArguments(args);
82 
83  for (auto& p : params_)
84  {
85  const string& name = p->getName().name();
86  const Argument *arg = namedArguments[name];
87 
88  if (not arg and not p->defaultValue())
89  throw SemanticException(
90  "missing argument to '" + name + "'", src);
91 
92  if (arg and not arg->type().isSubtype(p->type()))
93  throw WrongTypeException(p->type(),
94  arg->type(), arg->source());
95  }
96 }

118 
119  for (string argName : args)

124  throw SyntaxError(

127 
128  if (nextParameter == params_.end())

133  argName = p.getName().name();
