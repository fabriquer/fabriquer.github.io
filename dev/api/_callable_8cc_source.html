84 void Callable::CheckArguments(const ValueMap& args,
85  const StringMap<SourceRange>& argLocations,
86  const SourceRange& callLocation) const
87 {
88  for (auto& p : parameters_)
89  {
90  const string& name = p->name();
91  const Type& t = p->type();
92 
93  auto i = args.find(name);
94  if (i == args.end())
95  {
96  // Some arguments are optional.
97  if (p->defaultValue())
98  continue;
99 
100  throw SemanticException(
101  "missing argument to '" + name + "'", callLocation);
102  }
103 
104  const ValuePtr& arg = i->second;
105  assert(arg);
106 
107  if (not arg->type().isSubtype(t))
108  {
109  const SourceRange& argSource(argLocations.find(name)->second);
110  throw WrongTypeException(t, arg->type(), argSource);
111  }
112  }
113 }

116 vector<string>
117 Callable::NameArguments(const vector<string>& args, SourceRange src) const
118 {
119  vector<string> namedArgs;
120 
121  Bytestream& dbg = Bytestream::Debug("parser.callable");
122 
123  dbg << "matching arguments:\n ";
124  for (auto& a : args)
125  dbg << " " << (a.empty() ? "<unnamed>" : a);
126 
127  dbg << "\n to parameters:\n ";
128  for (auto& p : parameters_)
129  {
130  assert(p);
131  dbg << " "
132  << Bytestream::Definition << p->name()
133  << Bytestream::Operator << ":"
134  << p->type()
135  << Bytestream::Reset
136  ;
137  }
138 
139  dbg << "\n";
140 
141  bool doneWithPositionalArgs = false;
142  auto nextParameter = parameters_.begin();
143 
144  for (string argName : args)
145  {
146  if (argName.empty())
147  {
148  if (doneWithPositionalArgs)
149  throw SyntaxError(
150  "positional argument after keywords",
151  src);
152 
153  if (nextParameter == parameters_.end())
154  throw SyntaxError(
155  "too many positional arguments", src);
156 
157  const Parameter& p = **nextParameter++;
158  argName = p.name();
159  }
160  else
161  doneWithPositionalArgs = true;
162 
163  namedArgs.push_back(argName);
164  }
165 
166  return namedArgs;
167 }
