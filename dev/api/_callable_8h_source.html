77  template<class T>
78  StringMap<const T*> NameArguments(const UniqPtrVec<T>& v) const
79  {
80  std::vector<std::string> names;
81  SourceLocation begin, end;
82 
83  // Build a vector of what we currently know about the
84  // arguments' names.
85  for (auto& i : v)
86  {
87  const SourceRange& src = i->source();
88  if (not begin) begin = src.begin;
89  end = src.end;
90 
91  names.push_back(
92  i->hasName() ? i->getName().name() : "");
93  }
94 
95  // Fill in any gaps with knowledge about formal parameters.
96  names = NameArguments(names, SourceRange(begin, end));
97 
98  // Return the result.
99  StringMap<const T*> result;
100  for (size_t i = 0; i < v.size(); i++)
101  result.emplace(names[i], v[i].get());
102 
103  return result;
104  }
