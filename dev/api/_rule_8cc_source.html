104 void Rule::PrettyPrint(Bytestream& out, size_t /*indent*/) const
105 {
106  out
107  << Bytestream::Action << command_
108  << Bytestream::Operator << " {"
109  << Bytestream::Literal << " '" << description_ << "'"
110  ;
111 
112  for (auto& i : arguments_)
113  {
114  out
115  << Bytestream::Operator << ", "
116  << Bytestream::Definition << i.first
117  << Bytestream::Operator << " = "
118  << Bytestream::Literal << "'" << *i.second << "'"
119  ;
120  }
121 
122  if (not parameters().empty())
123  {
124  out << Bytestream::Operator << " <-";
125  for (auto& i : parameters())
126  out << " " << *i << Bytestream::Operator << ",";
127  }
128 
129  out
130  << Bytestream::Operator << " }"
131  << Bytestream::Reset
132  ;
133 }
