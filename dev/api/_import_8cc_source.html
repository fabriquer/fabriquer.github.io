104 dag::ValuePtr Import::evaluate(EvalContext& ctx) const
105 {
106  auto scope(ctx.EnterScope("import()"));
107 
108  dag::ValuePtr subdir(
109  ctx.builder().File(subdirectory(), dag::ValueMap(),
110  type().context().fileType(), source()));
111 
112  scope.set(ast::Subdirectory, subdir);
113 
114  // Gather arguments into an 'args' structure.
115  std::vector<dag::Structure::NamedValue> args;
116  for (const UniqPtr<ast::Argument>& a : arguments())
117  {
118  assert(a->hasName());
119 
120  const std::string& argName = a->getName().name();
121  dag::ValuePtr value = a->evaluate(ctx);
122 
123  args.emplace_back(argName, value);
124  }
125 
126  dag::DAGBuilder builder(ctx.builder());
127  if (plugin_)
128  return plugin_->Create(builder);
129 
130  dag::ValuePtr argStruct(
131  builder.Struct(args, this->scope().arguments(), source()));
132 
133  scope.set(ast::Arguments, argStruct);
134 
135  for (const UniqPtr<ast::Value>& v : this->scope().values())
136  v->evaluate(ctx);
137 
138  std::vector<dag::Structure::NamedValue> values;
139  Type::NamedTypeVec types;
140  for (auto& i : scope.leave())
141  {
142  values.emplace_back(i.first, i.second);
143  types.emplace_back(i.first, i.second->type());
144  }
145 
146  return builder.Struct(values, type(), source());
147 }
