93 ValuePtr List::Add(ValuePtr& n) const
94 {
95  SourceRange loc = SourceRange::Over(this, n.get());
96 
97  const List *next = n->asList();
98  if (not next)
99  throw SemanticException(
100  "lists can only be concatenated with lists", loc);
101 
102  if (not elementType_.isSupertype(next->elementType_)
103  and not next->elementType_.isSupertype(elementType_))
104  throw SemanticException(
105  "incompatible operands to concatenate (types "
106  + type().str() + ", " + next->type().str() + ")", loc);
107 
108  SharedPtrVec<Value> values(elements_);
109  auto& nextElem = next->elements_;
110  values.insert(values.end(), nextElem.begin(), nextElem.end());
111 
112  return ValuePtr(List::of(values, loc, elementType_.context()));
113 }
