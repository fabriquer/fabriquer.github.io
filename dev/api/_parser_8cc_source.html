164 Call* Parser::CreateCall(UniqPtr<Identifier>& name,
165  UniqPtr<UniqPtrVec<Argument>>& args,
166  const SourceRange& end)
167 {
168  if (not name or not args)
169  return nullptr;
170 
171  SourceRange src(name->source(), end);
172 
173  UniqPtr<SymbolReference> ref(Reference(std::move(name)));
174  if (not ref)
175  {
176  ReportError("call to undefined action/function", src);
177  return nullptr;
178  }
179 
180  // Run the arguments through Callable::CheckArguments to validate
181  // various rules (e.g. no positional arguments after keyword arguments).
182  auto& callable = dynamic_cast<const Callable&>(ref->definition());
183  try { callable.CheckArguments(*args, src); }
184  catch (SourceCodeException& e)
185  {
186  ReportError(e.message(), e.source());
187  return nullptr;
188  }
189 
190  auto& fnType = dynamic_cast<const FunctionType&>(ref->type());
191  return new Call(ref, *args, fnType.returnType(), src);
192 }

255 Function* Parser::DefineFunction(const SourceRange& begin,
256  UniqPtr<UniqPtrVec<Parameter>>& params,
257  UniqPtr<CompoundExpression>& body,
258  const Type *resultType)
259 {
260  if (not params or not body)
261  return nullptr;
262

264  {
265  ReportError(
266  "wrong return type ("

269  return nullptr;

271 
272  SourceRange loc(begin, *body);
273 
274  PtrVec<Type> parameterTypes;
275  for (auto& p : *params)

279 
280  const FunctionType *ty = ctx_.functionType(parameterTypes, *resultType);
281  return new Function(*params, *ty, body, loc);

361 Parameter* Parser::Param(UniqPtr<Identifier>&& name,
362  UniqPtr<Expression>&& defaultValue)
363 {
364  if (not name)
365  return nullptr;
366 
367  const Type *nameType = name->type();
368 
369  if (not name->isTyped() and not defaultValue)
370  {
371  ReportError("expected type or default value", *name);
372  return nullptr;
373  }
374 
375  if (defaultValue != nullptr and name->isTyped()
376  and !defaultValue->type().isSupertype(*name->type()))
377  {
378  ReportError("type mismatch", *defaultValue);
379  return nullptr;
380  }
381 
382  const Type& resultType = nameType ? *nameType : defaultValue->type();
383 
384  auto *p = new Parameter(name, resultType, std::move(defaultValue));
385  CurrentScope().Register(p);
386 
387  return p;

419 bool Parser::DefineValue(UniqPtr<Identifier>& id, UniqPtr<Expression>& e)
420 {
421  if (not id or not e)
422  return false;
423 
424  auto& scope(CurrentScope());
425 
426  if (scope.contains(*id))
427  {
428  ReportError("redefining value", *id);
429  return false;
430  }
